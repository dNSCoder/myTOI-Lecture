Chapter 01

อ่าน Input → Adjacency List

⏱ ~20 min · Pattern พื้นฐานที่สุด ต้องทำได้ทันที

โจทย์ CP ส่วนใหญ่ให้ input เป็นรูปแบบนี้:

Standard Input Format
7 1 2 1 3 1 4 2 5 2 6 4 7

บรรทัดแรกคือจำนวน node n ตามด้วย n-1 บรรทัดแสดง edge — เราแปลงเป็น adjacency list ได้ทันที:

C++pattern_input.cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;

    vector<vector<int>> adj(n + 1);  // index 1..n

    for (int i = 0; i < n - 1; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);   // undirected tree
        adj[v].push_back(u);
    }

    // แสดง adjacency list
    for (int u = 1; u <= n; u++) {
        cout << "Node " << u << ": ";
        for (int v : adj[u])
            cout << v << " ";
        cout << "\n";
    }
}
⚡ สิ่งสำคัญ

ใช้ adj(n+1) ไม่ใช่ adj(n) — เพราะ node นับจาก 1 ถึง n
Tree undirected → ต้อง push ทั้ง 2 ทิศทาง (adj[u]←v และ adj[v]←u)

Global Array แบบ CP จริง

ถ้าขนาด n ใหญ่มาก หรืออยากเร็วกว่า ใช้ global array แทน vector ใน main:

C++
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
vector<int> adj[MAXN];  // global — init เป็น empty vector อัตโนมัติ

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
}
Chapter 02

DFS Recursive บน Tree

⏱ ~20 min

DFS = Depth First Search — เดินลึกลงไปในแต่ละ subtree ก่อน แล้วค่อย backtrack กลับ

💡 Key Insight

บน Tree (ไม่มี cycle) ไม่ต้องเช็ค visited ก็ได้ ถ้าส่ง parent มาด้วย — ไม่เดินย้อนกลับขึ้นไปหา parent เอง

C++dfs_recursive.cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
vector<int> adj[MAXN];

// u = node ปัจจุบัน, p = parent (ไม่ให้เดินกลับ)
void dfs(int u, int p) {
    cout << u << " ";           // ทำอะไรกับ node นี้

    for (int v : adj[u]) {
        if (v == p) continue;  // ไม่เดินกลับไปหา parent
        dfs(v, u);
    }
}

int main() {
    int n;
    cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    dfs(1, 0);  // root = 1, parent = 0 (ไม่มี)
    cout << "\n";
}

ถ้าต้องการใช้ visited array แทน (เช่น ต้องใช้กับ graph ทั่วไปด้วย):

C++
vector<bool> visited(n + 1, false);

void dfs(int u) {
    visited[u] = true;
    cout << u << " ";
    for (int v : adj[u])
        if (!visited[v])
            dfs(v);
}
⚠️ Stack Overflow

Recursive DFS บน tree ที่ลึก ~10⁵ nodes จะ stack overflow — ถ้าโจทย์ให้ n ใหญ่มาก ให้ใช้ DFS Iterative ใน Ch3 แทน

Chapter 03

DFS Iterative ด้วย Stack

⏱ ~25 min · สำคัญมาก — แก้ปัญหา stack overflow

แทน call stack ด้วย stack<int> ของเราเอง — ผลลัพธ์เหมือน DFS recursive

C++dfs_iterative.cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
vector<int> adj[MAXN];

void dfs(int root, int n) {
    vector<bool> visited(n + 1, false);
    stack<int> st;

    st.push(root);
    visited[root] = true;

    while (!st.empty()) {
        int u = st.top();
        st.pop();

        cout << u << " ";  // ทำอะไรกับ u

        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                st.push(v);
            }
        }
    }
    cout << "\n";
}

int main() {
    int n; cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    dfs(1, n);
}
⚡ Recursive vs Iterative

ลำดับ visit อาจต่างกันเล็กน้อย (ขึ้นกับลำดับ push) แต่ครอบคลุม node ครบเท่ากัน — ถ้าต้องการลำดับเหมือนกัน ให้ push children ย้อนลำดับ

เปรียบเทียบ

วิธีข้อดีข้อเสีย
Recursiveเขียนสั้น อ่านง่ายStack overflow ถ้า n ใหญ่
Iterativeไม่มี stack overflowเขียนยาวกว่า
Chapter 04

BFS ด้วย Queue

⏱ ~20 min · เดินทีละ Level

BFS = Breadth First Search — เดินทีละชั้น ใช้ queue เข้าแถวทีละ node

C++bfs.cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
vector<int> adj[MAXN];

void bfs(int root, int n) {
    vector<bool> visited(n + 1, false);
    queue<int> q;

    q.push(root);
    visited[root] = true;

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        cout << u << " ";  // เยี่ยม u

        for (int v : adj[u]) {
            if (!visited[v]) {
                visited[v] = true;
                q.push(v);
            }
        }
    }
    cout << "\n";
}

int main() {
    int n; cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    bfs(1, n);
}
⚡ เมื่อไรใช้ BFS แทน DFS

หาระยะ shortest path ใน unweighted graph/tree — BFS รับประกัน visit node ตามลำดับระยะทาง (level)
หาว่า node ไหนอยู่ level เดียวกัน — BFS ทำได้ตรงๆ

Chapter 05

Pattern: depth[] & parent[]

⏱ ~20 min · Pattern ที่ออกบ่อยมากใน Olympiad

ต่อยอดจาก DFS/BFS — คำนวณ depth และ parent ของทุก node ไว้ใน array เพื่อ query ทีหลัง

C++depth_parent.cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
vector<int> adj[MAXN];
int dep[MAXN];    // depth ของแต่ละ node
int par[MAXN];    // parent ของแต่ละ node

void dfs(int u, int p, int d) {
    par[u] = p;
    dep[u] = d;

    for (int v : adj[u]) {
        if (v == p) continue;
        dfs(v, u, d + 1);
    }
}

int main() {
    int n; cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    dfs(1, 0, 0);  // root=1, parent=0, depth=0

    for (int u = 1; u <= n; u++)
        cout << "node " << u
             << " depth=" << dep[u]
             << " parent=" << par[u] << "\n";
}

Output สำหรับ tree ตัวอย่าง (root=1):

Output
node 1 depth=0 parent=0 node 2 depth=1 parent=1 node 3 depth=1 parent=1 node 4 depth=1 parent=1 node 5 depth=2 parent=2 node 6 depth=2 parent=2 node 7 depth=2 parent=4
⚡ ใช้ทำอะไรได้บ้าง

หา depth ของ node ใดก็ได้ใน O(1) หลัง DFS
หา parent ของ node ใดก็ได้ใน O(1)
เป็นฐานของ LCA (Lowest Common Ancestor) ซึ่งต่อยอดใน Binary Lifting

Chapter 06

Pattern: Subtree Size

⏱ ~20 min · หัวใจของ Tree DP

นับจำนวน node ใน subtree ของแต่ละ node — เป็น pattern พื้นฐานของ Tree DP ทั้งหมด

แนวคิด: sz[u] = 1 + ผลรวม sz[v] ของทุก child v

C++subtree_size.cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 1e5 + 5;
vector<int> adj[MAXN];
int sz[MAXN];     // subtree size ของแต่ละ node

void dfs(int u, int p) {
    sz[u] = 1;   // นับตัวเอง

    for (int v : adj[u]) {
        if (v == p) continue;
        dfs(v, u);
        sz[u] += sz[v];   // postorder: ลูกเสร็จแล้ว บวกเข้ามา
    }
}

int main() {
    int n; cin >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    dfs(1, 0);

    for (int u = 1; u <= n; u++)
        cout << "sz[" << u << "] = " << sz[u] << "\n";
}
⚡ ทำไม Postorder

ต้องรู้ sz ของลูกก่อนจึงจะบวกได้ — นี่คือเหตุผลที่ Postorder (ลูกก่อน parent) สำคัญมากใน Tree DP

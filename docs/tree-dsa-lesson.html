Chapter 01 · ~20 min

Tree คืออะไร?

🕐 ประมาณ 20 นาที

Tree คือโครงสร้างข้อมูลแบบ Nonlinear Hierarchical ประกอบด้วย Nodes เชื่อมกันด้วย Edges ข้อมูลจัดเป็นลำดับชั้น ไม่เหมือน Array หรือ Linked List ที่เก็บตามลำดับ

📖 นิยาม

Tree คือ Acyclic Graph ที่มี Node เดียวไม่มี parent เรียกว่า Root — ทุก Node อื่นเชื่อมถึง Root ได้ผ่าน Edge เสมอ และ Tree ที่มี n nodes มี n-1 edges เสมอ

🔑 Terminology ที่ต้องรู้

Node บนสุด ไม่มี Parent

Sibling

จำนวน Edge จาก Root ถึง Node

จำนวน Children ของ Node
Subtree

Ancestor
ทุก Node บนเส้นทางจาก Root

Edge
เส้นเชื่อม 2 Nodes — n nodes มี n-1 edges
Level
Depth ของ Node (Root = Level 0)
💡 Real-world Applications

File System (folder → subfolder → file) · DOM ใน HTML Browser · Organizational Chart · Expression Parser · Routing Algorithms

Chapter 02 · ~25 min

Representation & Node Structure

🕐 ประมาณ 25 นาที

ก่อน Implement ต้องเข้าใจว่า Tree เก็บในหน่วยความจำอย่างไร วิธีหลักคือ Linked Nodes — แต่ละ Node เก็บ data และ pointer ไปหา children

📦 Generic Tree Node (N-ary)

C++
#include <iostream>
#include <vector>
using namespace std;

// Generic Tree Node — มี children ได้ไม่จำกัด (N-ary)
struct Node {
    int data;
    vector<Node*> children;

    // Constructor
    Node(int x) : data(x) {}
};

// เพิ่ม child ให้ parent
void addChild(Node* parent, Node* child) {
    parent->children.push_back(child);
}

int main() {
    Node* root = new Node(1);
    Node* n2   = new Node(2);
    Node* n3   = new Node(3);
    Node* n4   = new Node(4);

    addChild(root, n2);  // 1 → [2, 3]
    addChild(root, n3);
    addChild(n2, n4);    // 2 → [4]

    cout << root->data << endl;            // 1
    cout << root->children[0]->data;      // 2
    return 0;
}

🔢 Binary Tree Node

Binary Tree มี children ได้สูงสุด 2 ตัว ใช้ pointer left และ right

C++
#include <iostream>
using namespace std;

// Binary Tree Node
struct Node {
    int data;
    Node* left;
    Node* right;

    Node(int x) : data(x), left(nullptr), right(nullptr) {}
};

int main() {
    // สร้าง Binary Tree ด้วยมือ
    Node* root       = new Node(1);
    root->left        = new Node(2);
    root->right       = new Node(3);
    root->left->left  = new Node(4);
    root->left->right = new Node(5);

    /*  Tree ที่ได้:
             1
            / \
           2   3
          / \
         4   5
    */
    cout << root->left->right->data;  // 5
    return 0;
}

Chapter 03 · ~35 min

Tree Traversal

🕐 ประมาณ 35 นาที

Traversal คือการ "เดิน" ผ่านทุก Node — แบ่งเป็น DFS (Depth-First) ซึ่งมี 3 แบบ และ BFS (Level-Order)

Inorder — Left → Root → Right

เดิน Left ก่อน → Root → Right — ใน BST จะได้ผลที่ Sorted เสมอ

C++ · Inorder (Recursive)
#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* left;
    Node* right;
    Node(int x) : data(x), left(nullptr), right(nullptr) {}
};

void inorder(Node* node) {
    if (node == nullptr) return;

    inorder(node->left);           // 1. ไป Left ก่อน
    cout << node->data << " ";     // 2. เข้า Root
    inorder(node->right);          // 3. ไป Right
}

int main() {
    // BST: 4 root, 2 left, 6 right ...
    Node* root       = new Node(4);
    root->left        = new Node(2);
    root->right       = new Node(6);
    root->left->left  = new Node(1);
    root->left->right = new Node(3);
    root->right->left = new Node(5);
    root->right->right= new Node(7);

    inorder(root);
    // Output: 1 2 3 4 5 6 7  ← Sorted!
    return 0;
}
Preorder — Root → Left → Right

เข้า Root ก่อน → ลง Left → ลง Right — ใช้ Copy tree หรือ Serialize โครงสร้าง

C++ · Preorder (Recursive)
void preorder(Node* node) {
    if (node == nullptr) return;

    cout << node->data << " ";     // 1. เข้า Root ก่อน
    preorder(node->left);          // 2. ไป Left
    preorder(node->right);         // 3. ไป Right
}

// Tree:    4
//         / \
//        2   6
//       / \ / \
//      1  3 5  7
//
// Output: 4 2 1 3 6 5 7
Postorder — Left → Right → Root

ลงถึง Leaf ก่อน → ค่อยขึ้น Root สุดท้าย — ใช้ลบ Tree หรือคำนวณ Expression Tree

C++ · Postorder (Recursive)
void postorder(Node* node) {
    if (node == nullptr) return;

    postorder(node->left);         // 1. ไป Left
    postorder(node->right);        // 2. ไป Right
    cout << node->data << " ";     // 3. เข้า Root สุดท้าย
}

// Tree:    4
//         / \
//        2   6
//       / \ / \
//      1  3 5  7
//
// Output: 1 3 2 5 7 6 4
BFS / Level-Order Traversal

เดิน Level ต่อ Level จากบนลงล่าง — ใช้ Queue แทน Recursion

C++ · BFS (Level-Order)
#include <iostream>
#include <queue>
using namespace std;

void levelOrder(Node* root) {
    if (!root) return;

    queue<Node*> q;
    q.push(root);

    while (!q.empty()) {
        Node* curr = q.front();
        q.pop();

        cout << curr->data << " ";

        if (curr->left)  q.push(curr->left);
        if (curr->right) q.push(curr->right);
    }
}

// Tree:    4
//         / \
//        2   6
//       / \ / \
//      1  3 5  7
//
// Output: 4 2 6 1 3 5 7  ← Level by level

Binary Tree

🕐 ประมาณ 20 นาที

Binary Tree คือ Tree ที่แต่ละ Node มีลูกได้ สูงสุด 2 ตัว คือ left child และ right child เป็น Tree ที่พบบ่อยและสำคัญที่สุดในการเขียนโปรแกรม

Full Binary Tree

Full

Binary Search Tree

Self-Balancing

Chapter 05 · ~15 min

Full Binary Tree

🕐 ประมาณ 15 นาที

Full Binary Tree คือ Binary Tree ที่ทุก Node มีลูก 0 หรือ 2 ตัว เท่านั้น

📐 Property

ถ้า Full BT มี L Leaf Nodes → มี L−1 Internal Nodes เสมอ — Total nodes = 2L−1

C++ · ตรวจสอบ Full Binary Tree
bool isFullBinaryTree(Node* node) {
    // Leaf node = full
    if (node == nullptr) return true;
    if (!node->left && !node->right) return true;

    // มีลูกทั้ง 2 ข้าง → ตรวจ subtree ต่อ
    if (node->left && node->right)
        return isFullBinaryTree(node->left)
            && isFullBinaryTree(node->right);

    // มีลูกข้างเดียว = ไม่ใช่ Full
    return false;
}
Chapter 06 · ~15 min

Perfect Binary Tree

🕐 ประมาณ 15 นาที

Perfect Binary Tree คือ Binary Tree ที่ ทุก Level เต็มหมด และทุก Leaf อยู่ที่ Depth เดียวกัน — เป็นทั้ง Full และ Complete BT ในเวลาเดียวกัน

📐 Properties

Perfect BT height h: Nodes ทั้งหมด = 2ʰ⁺¹ − 1 · Leaf = 2ʰ · Internal = 2ʰ − 1

C++ · ตรวจสอบ Perfect Binary Tree
int height(Node* node) {
    int h = 0;
    while (node) { h++; node = node->left; }
    return h;
}

bool isPerfect(Node* node, int depth, int level = 0) {
    if (!node) return true;

    // Leaf: ต้องอยู่ที่ depth เดิม
    if (!node->left && !node->right)
        return depth == level + 1;

    // ต้องมีลูกครบ 2 ข้าง
    if (!node->left || !node->right) return false;

    return isPerfect(node->left,  depth, level + 1)
        && isPerfect(node->right, depth, level + 1);
}

// ใช้งาน:
int d = height(root);
cout << (isPerfect(root, d) ? "Perfect" : "Not Perfect");

Complete Binary Tree

🕐 ประมาณ 15 นาที

Complete Binary Tree คือ Binary Tree ที่ ทุก Level เต็มยกเว้น Level สุดท้าย ซึ่งต้องเติม Node จาก ซ้ายไปขวา

💡 ทำไมสำคัญ?

Complete BT เป็นพื้นฐานของ Binary Heap (Priority Queue & Heap Sort) และ Represent ด้วย Array ได้โดยไม่ต้องใช้ Pointer

📦 Array Representation

C++ · Array Representation of Complete BT
#include <iostream>
#include <vector>
using namespace std;

// Complete BT เก็บใน Array:
// index:  0  1  2  3  4  5
// value: [1, 2, 3, 4, 5, 6]
//
//         1 (i=0)
//        / \
//       2   3  (i=1, 2)
//      / \ /
//     4  5 6   (i=3, 4, 5)

int parent(int i)     { return (i - 1) / 2; }
int leftChild(int i) { return 2 * i + 1; }
int rightChild(int i){ return 2 * i + 2; }

int main() {
    vector<int> tree = {1, 2, 3, 4, 5, 6};

    cout << parent(3)      << endl; // 1 (node 4's parent is node 2)
    cout << leftChild(1)  << endl; // 3 (index of left child of node 2)
    cout << rightChild(1) << endl; // 4 (index of right child of node 2)
    return 0;
}
Chapter 08 · ~15 min

Balanced Binary Tree

🕐 ประมาณ 15 นาที

Balanced Binary Tree คือ Binary Tree ที่ |height(left) − height(right)| ≤ 1 ที่ ทุก Node ทำให้มั่นใจได้ว่า Search = O(log n)

⚠️ Unbalanced Tree Problem

BST ที่ Insert ข้อมูล Sorted เช่น 1,2,3,4,5 จะ Skewed กลายเป็น Linked List — Search กลายเป็น O(n) แทน O(log n)

C++ · ตรวจสอบ Balanced Binary Tree
// คืนค่า height ถ้า balanced, -1 ถ้าไม่ balanced
int checkBalance(Node* node) {
    if (!node) return 0;

    int lh = checkBalance(node->left);
    if (lh == -1) return -1;

    int rh = checkBalance(node->right);
    if (rh == -1) return -1;

    // Balance Factor
    if (abs(lh - rh) > 1) return -1;

    return max(lh, rh) + 1;
}

bool isBalanced(Node* root) {
    return checkBalance(root) != -1;
}

// ใช้งาน:
cout << (isBalanced(root) ? "Balanced" : "Not Balanced");
Chapter 09 · ~30 min

Binary Search Tree (BST)

🕐 ประมาณ 30 นาที

BST คือ Binary Tree ที่มีกฎ: Left Subtree < Node < Right Subtree ที่ทุก Node ทำให้ Search, Insert, Delete ได้ O(log n) เฉลี่ย

🔍 BST — Insert, Search, Delete

C++ · BST Insert + Search + Delete
#include <iostream>
using namespace std;

struct Node {
    int data;
    Node* left;
    Node* right;
    Node(int x) : data(x), left(nullptr), right(nullptr) {}
};

// ── INSERT ── O(log n) average
Node* insert(Node* node, int key) {
    if (!node) return new Node(key);
    if (key < node->data)
        node->left  = insert(node->left,  key);
    else if (key > node->data)
        node->right = insert(node->right, key);
    return node;  // duplicate → ignore
}

// ── SEARCH ── O(log n) average
Node* search(Node* node, int key) {
    if (!node || node->data == key) return node;
    return key < node->data
        ? search(node->left,  key)
        : search(node->right, key);
}

// ── Helper: หา node ที่มีค่าน้อยสุด ──
Node* minNode(Node* node) {
    while (node->left) node = node->left;
    return node;
}

// ── DELETE ── 3 กรณี
Node* deleteNode(Node* node, int key) {
    if (!node) return nullptr;

    if      (key < node->data) node->left  = deleteNode(node->left,  key);
    else if (key > node->data) node->right = deleteNode(node->right, key);
    else {
        // Case 1: ไม่มี left child
        if (!node->left) {
            Node* temp = node->right;
            delete node;
            return temp;
        }
        // Case 2: ไม่มี right child
        if (!node->right) {
            Node* temp = node->left;
            delete node;
            return temp;
        }
        // Case 3: มีลูก 2 ข้าง → แทนด้วย inorder successor
        Node* successor = minNode(node->right);
        node->data      = successor->data;
        node->right     = deleteNode(node->right, successor->data);
    }
    return node;
}

int main() {
    Node* root = nullptr;
    for (int v : {8, 3, 10, 1, 6, 9, 14})
        root = insert(root, v);

    Node* found = search(root, 6);
    cout << (found ? "Found: " : "Not found");
    if (found) cout << found->data;  // Found: 6

    root = deleteNode(root, 3);
    cout << (search(root, 3) ? "Found" : "Deleted"); // Deleted
    return 0;
}

Chapter 10 · ~25 min

AVL Tree

🕐 ประมาณ 25 นาที

AVL Tree (Adelson-Velsky & Landis, 1962) คือ Self-Balancing BST ที่ Balance Factor ทุก Node ต้องอยู่ใน {−1, 0, 1} — เมื่อ Insert/Delete แล้วไม่ balanced จะทำ Rotation อัตโนมัติ

📐 Balance Factor

BF(node) = height(left) − height(right) — ถ้า |BF| > 1 ต้อง Rotate ทันที

🔄 4 ประเภทของ Rotation

Caseเงื่อนไขการแก้ไขตัวอย่าง Insert
LLBF > 1 && key < leftRight Rotation30 → 20 → 10
RRBF < -1 && key > rightLeft Rotation10 → 20 → 30
LRBF > 1 && key > leftLeft then Right30 → 10 → 20
RLBF < -1 && key < rightRight then Left10 → 30 → 20
C++ · AVL Tree Full Implementation
#include <iostream>
#include <algorithm>
using namespace std;

struct Node {
    int data, height;
    Node* left;
    Node* right;
    Node(int x) : data(x), height(1), left(nullptr), right(nullptr) {}
};

int h(Node* n)  { return n ? n->height : 0; }
int bf(Node* n) { return n ? h(n->left) - h(n->right) : 0; }

void updateH(Node* n) {
    n->height = 1 + max(h(n->left), h(n->right));
}

// ── Rotations ──
Node* rotateRight(Node* y) {
    Node* x   = y->left;
    Node* T2  = x->right;
    x->right  = y;
    y->left   = T2;
    updateH(y); updateH(x);
    return x;  // x กลายเป็น root ใหม่
}

Node* rotateLeft(Node* x) {
    Node* y   = x->right;
    Node* T2  = y->left;
    y->left   = x;
    x->right  = T2;
    updateH(x); updateH(y);
    return y;  // y กลายเป็น root ใหม่
}

// ── Balance หลัง Insert/Delete ──
Node* balance(Node* node) {
    updateH(node);
    int b = bf(node);

    if (b > 1) {                         // Left heavy
        if (bf(node->left) < 0)           // LR case
            node->left = rotateLeft(node->left);
        return rotateRight(node);         // LL case
    }
    if (b < -1) {                        // Right heavy
        if (bf(node->right) > 0)          // RL case
            node->right = rotateRight(node->right);
        return rotateLeft(node);          // RR case
    }
    return node;
}

// ── INSERT ──
Node* insert(Node* node, int key) {
    if (!node) return new Node(key);
    if      (key < node->data) node->left  = insert(node->left,  key);
    else if (key > node->data) node->right = insert(node->right, key);
    return balance(node);  // auto-balance หลัง insert
}

int main() {
    Node* root = nullptr;

    // RR case: 10, 20, 30 → auto Left Rotation
    root = insert(root, 10);
    root = insert(root, 20);
    root = insert(root, 30);  // triggers Left Rotation

    cout << root->data;  // → 20 (root ถูก balance แล้ว)
    cout << root->left->data;   // → 10
    cout << root->right->data;  // → 30
    return 0;
}

🔑 AVL vs BST สรุป

BST ธรรมดา: O(log n) เฉลี่ย แต่ O(n) worst case (Skewed) — AVL Tree: รับประกัน O(log n) ทุกกรณี เพราะ height ≤ 1.44 log₂(n+2)
